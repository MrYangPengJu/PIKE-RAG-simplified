# 二叉树

```C
#include <stdio.h>
#include <stdlib.h>
#define ElemType int
// 定义二叉树结点结构
typedef struct TreeNode {
    ElemType val;
    struct TreeNode *left;
    struct TreeNode *right;
} TreeNode;

// 创建二叉树
TreeNode* createTree(ElemType val) {
    TreeNode* node = (TreeNode*)malloc(sizeof(TreeNode));
    node->val = val;
    node->left = NULL;
    node->right = NULL;
    return node;
}

// 前序遍历
void preOrder(TreeNode* root) {
    if (root == NULL) {
        return;
    }
    printf("%d ", root->val);
    preOrder(root->left);
    preOrder(root->right);
}

// 中序遍历
void inOrder(TreeNode* root) {
    if (root == NULL) {
        return;
    }
    inOrder(root->left);
    printf("%d ", root->val);
    inOrder(root->right);
}

// 后序遍历
void postOrder(TreeNode* root) {
    if (root == NULL) {
        return;
    }
    postOrder(root->left);
    postOrder(root->right);
    printf("%d ", root->val);
}

int main() {
    // 创建二叉树
    TreeNode* root = createTree(1);
    root->left = createTree(2);
    root->right = createTree(3);
    root->left->left = createTree(4);
    root->left->right = createTree(5);

    // 前序遍历
    printf("PreOrder: ");
    preOrder(root);
    printf("\n");

    // 中序遍历
    printf("InOrder: ");
    inOrder(root);
    printf("\n");

    // 后序遍历
    printf("PostOrder: ");
    postOrder(root);
    printf("\n");

    return 0;
}

```
