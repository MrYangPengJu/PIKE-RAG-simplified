# 栈

## 基于数组的栈
```C
#include <stdio.h>
#include <stdlib.h>

#define MAX_SIZE 1000
#define ElemType int

typedef struct Stack {
	ElemType data[MAX_SIZE];
	int top;
}Stack;

Stack* init()
{
	Stack* S = (Stack*)malloc(sizeof(Stack));
	S->top = -1;
	return S;
}
int isEmpty(Stack* S)
{
	return S->top == -1;
}
ElemType pop(Stack* S)
{
	if (isEmpty(S))
		return false;
	return S->data[S->top--];
}
bool push(Stack* S, ElemType value)
{
	if (S->top > MAX_SIZE)
		return false;
	S->data[++S->top] = value;
	return true;
}
ElemType checkTop(Stack* S)
{
	return S->data[S->top];
}

```

## 基于链表的栈
```C
#include <stdio.h>
#include <stdlib.h>
#define ElemType int

typedef struct Stack {
	struct Stack* next;
	ElemType data;
}Stack;

Stack* stackInit()
{
	Stack* S = (Stack*)malloc(sizeof(Stack));
	S->next = NULL;
	return S;
}
bool isEmpty(Stack* S)
{
	return S->next == NULL;
}
bool push(Stack* S, ElemType value)
{
	Stack* node = (Stack*)malloc(sizeof(Stack));
	node->data = value;
	node->next = S->next;
	S->next = node;
	return true;
}
ElemType pop(Stack* S)
{
	if (isEmpty(S))
	{
		return false;
	}
	Stack* tp = S->next;
	ElemType ret = tp->data;
	S->next = tp->next;
	free(tp);
	return ret;
}
```
